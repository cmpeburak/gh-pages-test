 29def filter_unsupported_kwargs(logger: Optional[Callable] = None):
 30    """Decorator for filtering out kwargs which are not supported.
 31
 32    Args:
 33        logger: A callable that takes a string and logs it. In this case, it will
 34            only be used to warn about any kwargs that will be filtered out due to
 35            not being supported by the wrapped function.
 36
 37    """
 38
 39    def remove_wrong_kwargs_(func):
 40        func_kwargs = set(get_callable_kwargs(func))
 41
 42        @wraps(func)
 43        def func_(*args, **kwargs):
 44            unsupported_kwargs = {
 45                k: v for k, v in kwargs.items() if k not in func_kwargs
 46            }
 47            if unsupported_kwargs:
 48                kwargs = {k: v for k, v in kwargs.items() if k in func_kwargs}
 49                if logger:
 50                    logger(f"WARNING: filtering out {unsupported_kwargs}")
 51            return func(*args, **kwargs)
 52
 53        return func_
 54
 55    return remove_wrong_kwargs_

 64def retry(
 65    max_tries: int = 3,
 66    wait_secs: int = 5,
 67    raise_on_fail: bool = True,
 68    res_on_fail: Any = None,
 69) -> Any:
 70    """Decorator for allowing a function to retry
 71
 72    Args:
 73        max_tries: Maximum number of attempts to call the function. (default: 1)
 74        wait_secs: time to wait between failures in seconds. (default: 5)
 75        raise_on_fail: Whether to raise if all attempts fail (default: True)
 76        res_on_fail: value to return in case of failure (default: None). Obviously,
 77            this only makes sense when raise_on_fail is set to False.
 78    """
 79
 80    def retry_(func):
 81        @wraps(func)
 82        def retry_func(*args, **kwargs):
 83            tries = 0
 84            while True:
 85                try:
 86                    return func(*args, **kwargs)
 87                except Exception as e:
 88                    tries += 1
 89                    print(f"ERROR: attempt={tries} failed in calling {func}: {e}")
 90                    if tries >= max_tries:
 91                        print(f"Max tries={max_tries} reached for {func}.")
 92                        if raise_on_fail:
 93                            raise e
 94                        else:
 95                            break
 96                    time.sleep(wait_secs)
 97                    print(f"Retrying to call {func}")
 98            return res_on_fail
 99
100        return retry_func
101
102    return retry_

105def diff_rec(
106    left: Union[dict, Iterable],
107    right: Union[dict, Iterable],
108    ignore_keys: Optional[set] = None,
109    comp_mode: str = "dp",
110    precision: int = 6,
111) -> Union[Dict, Tuple]:
112    """Returns diff of two dictionaries (or non-string iterables).
113    It uses recursive drill down, so do not use on structures which have big depth!
114
115    Args:
116        left: left dict or non-string iterable
117        right: right dict or non-string iterable
118        ignore_keys: optional set of keys to be ignored during comparison
119        comp_mode: Numerical comparison modes. Must be one of
120            - sf : significant figures
121            - dp : decimal places
122            - pc : percentage change
123        precision: precision depending on the mode:
124            if comp_mode='sf' then significant figures to round to before comparison
125            if comp_mode='dp' then decimal places to round to before comparison
126            if comp_mode='pc' then percentage change required to trigger diff
127
128    Returns:
129        diff of left and right. If left and right were dictionaries that are not
130        identical, then one or more of the following keys will exist:
131            - "ValDiff": if there are keys which have value differences, then this
132                will contain a dictionary of those keys with values being tuple of
133                (left_value, right_value, diff_value) if numerical or simply
134                (left_value, right_value) if not numerical.
135            - "OnlyInLeft": if there are some keys that only exist in the left, then
136                those keys and their values will appear here.
137            - "OnlyInLeft": if there are some keys that only exist in the right, then
138                those keys and their values will appear here.
139    """
140    if ignore_keys is None:
141        ignore_keys = set()
142    ignore_keys = set(ignore_keys)
143
144    def diff_rec_(left, right):
145        diffs = {}
146        left_is_dict = isinstance(left, dict)
147        right_is_dict = isinstance(right, dict)
148        if left_is_dict and right_is_dict:
149            left_keys = set(left)
150            right_keys = set(right)
151            shared_keys = left_keys & right_keys - ignore_keys
152            only_in_left_keys = left_keys - right_keys - ignore_keys
153            only_in_right_keys = right_keys - left_keys - ignore_keys
154            map_diffs = {}
155            for key in shared_keys:
156                res = diff_rec_(left[key], right[key])
157                if res:
158                    if "ValDiff" not in map_diffs:
159                        map_diffs["ValDiff"] = {}
160                    map_diffs["ValDiff"][key] = res
161            if only_in_left_keys:
162                tmp_map = {}
163                for k in only_in_left_keys:
164                    tmp_map[k] = left[k]
165                map_diffs["OnlyInLeft"] = tmp_map
166            if only_in_right_keys:
167                tmp_map = {}
168                for k in only_in_right_keys:
169                    tmp_map[k] = right[k]
170                map_diffs["OnlyInRight"] = tmp_map
171            if map_diffs:
172                diffs = map_diffs
173        elif not left_is_dict and not right_is_dict:
174            left_is_iterable = hasattr(left, "__iter__") and not isinstance(left, str)
175            right_is_iterable = hasattr(right, "__iter__") and not isinstance(
176                right, str
177            )
178            if left_is_iterable and right_is_iterable:
179                iterable_diffs = {}
180                idx = -1
181                for item1, item2 in zip_longest(left, right):
182                    idx += 1
183                    res = diff_rec_(item1, item2)
184                    if res:
185                        iterable_diffs[idx] = res
186                if iterable_diffs:
187                    diffs = iterable_diffs
188            elif not left_is_iterable and not right_is_iterable:
189                if left != right:
190                    if comp_mode == "sf":
191                        try:
192                            left_num = ("%%.%dg" % (int(precision))) % (left)
193                            right_num = ("%%.%dg" % (int(precision))) % (right)
194                            if left_num != right_num:
195                                diffs = (left, right, right - left)
196                        except:  # noqa: E722
197                            diffs = (left, right)
198                    elif comp_mode == "dp":
199                        try:
200                            left_num = ("{0:.%sf}" % (int(precision))).format(
201                                float(left)
202                            )
203                            right_num = ("{0:.%sf}" % (int(precision))).format(
204                                float(right)
205                            )
206                            if left_num != right_num:
207                                diffs = (left, right, right - left)
208                        except:  # noqa: E722
209                            diffs = (left, right)
210                    elif comp_mode == "pc":
211                        try:
212                            if abs(right - left) / float(abs(left)) >= precision:
213                                diffs = (left, right, right - left)
214                        except:  # noqa: E722
215                            diffs = (left, right)
216                    else:
217                        raise Exception("comp_mode='%s' not supported" % (comp_mode))
218            else:
219                diffs = (left, right)
220        else:
221            diffs = (left, right)
222        return diffs
223
224    return diff_rec_(left, right)

262def convert_args_list_to_dict(args_list: List[str]) -> Dict:
263    """Converts given arguments lists to a dictionary of argument name
264    and value pairs.
265
266    Args:
267       args_list: A list containing of script arguments.
268       It does not support boolean flags. All argument names should start with "--"
269       and should be followed by the argument value.
270       e.g. ['--test_value', 'A', '--num', '10']
271
272    Returns:
273        A dictionary of argument name and value pairs.
274        e.g. {"test_value": "A", "num": 10}
275    """
276    assert len(args_list) % 2 == 0, (
277        "The length of the list is not divisible by 2, "
278        "make sure that all arguments have name-value pairs."
279    )
280
281    args_dict = {}
282    for i in range(0, len(args_list), 2):
283        argument_name = args_list[i]
284        argument_value = args_list[i + 1]
285
286        assert argument_name.startswith(
287            "--"
288        ), "Argument name must start will double dashes."
289        assert not argument_value.startswith(
290            "--"
291        ), "Argument value must not start with double dashes."
292
293        if isint(argument_value):
294            argument_value = int(argument_value)
295        elif isfloat(argument_value):
296            argument_value = float(argument_value)
297
298        argument_name = argument_name[2:]  # Remove the dashes from the name
299        args_dict[argument_name] = argument_value
300
301    return args_dict
